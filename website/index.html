Developer API Access

import requests

# Analyze flight emissions
response = requests.post('https://api.contrailtracker.com/v1/analyze',
    json={
        'flight_data': 'path/to/flights.csv',
        'satellite_data': 'path/to/satellite.nc',
        'carbon_market': 'EU_ETS'
    },
    headers={'Authorization': 'Bearer YOUR_API_KEY'}
)

results = response.json()
print(f"Total CO2: {results['total_co2_tonnes']} tonnes")
print(f"Carbon Cost: ${results['carbon_cost_usd']:,.2f}")
print(f"Contrail Impact: +{results['contrail_impact_pct']}%")
